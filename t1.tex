    \section{#1}
        \setboolean{sectiontoc}{true}
    }

    \setbeamercolor{palette secondary}{bg=white,fg=UBCblue}
    \setbeamercolor{palette tertiary}{bg=UBCblue,fg=white}
    \setbeamercolor{palette quaternary}{bg=UBCblue,fg=white}
    \setbeamercolor{structure}{fg=UBCblue} % itemize, enumerate, etc
    \setbeamercolor{section in toc}{fg=UBCblue} % TOC sections
    \setbeamercolor{date in head/foot}{bg=white,fg=UBCblue}
    \setbeamercolor{title in head/foot}{bg=white,fg=UBCblue}
    
    % Override palette coloring with secondary
    \setbeamercolor{section in foot}{bg = white, fg=white}
    \setbeamercolor{subsection in head}{fg=white}
    \setbeamercolor{subsection in foot}{bg = white, fg=UBCblue}
    
    \setbeamerfont{title like}{shape=\scshape}
    \setbeamerfont{frametitle}{shape=\scshape}

                    % cuando no hay ninifrmaes (guia arriba)  headingThree = {\begin{frame}\frametitle{#1}},
